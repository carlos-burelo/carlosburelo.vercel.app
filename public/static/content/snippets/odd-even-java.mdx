---
title: Numeros pares e impares usando java
type: java
---

```java
import javax.swing.JOptionPane;

public class OddOrEven {
  public static void main(String[] args) {
    int number = Integer.parseInt(
      JOptionPane.showInputDialog(null, "INGRESE EL NUMERO")
    );
    if (number % 2 != 0) print(number, "impar");
    else print(number, "par");
  }
  static void print(int number, String type) {
    JOptionPane.showMessageDialog(null, "El numero " + number + " es " + type);
  }
}
```
